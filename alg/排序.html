希尔排序：是插入排序的一个变种。无论是插入排序还是冒泡排序，如果最大元在数组首部，则要移动n-1此才能到达排序位置。也就是说，如果原数组中一个元素距离其正确的位置很远时，采用插入和冒泡排序需要很多次操作才能使其到正确的位置，这样非常消耗时间。
而希尔排序就是为了加速排序而改进的插入排序，交换不相邻的元素来对数组进行局部排序。
希尔排序采用插入排序方法，先让数组中间隔为h=n/2的元素有序，再让间隔为h=n/4的元素有序，如此循环，当h=1时，就让间隔为1的元素有序，即整个数组有序。

C代码：
#include<stdio.h>

void sort(int [],int,int,int);
void xier(int arr[],int n)
{
    int k,head;
    for(k=n/2;k>0;k/=2)
    {
        for(head=0;head<k;head++)
        {
            sort(arr,head,k,n);
        }
    }
}

void sort(int arr[],int head,int k,int n)
{
    int i;
    for(i=head+k;i<n;i+=k)
    {
        int j=i;
        int val=arr[i];
        while(j>head && arr[j-k]>val)
        {
            arr[j]=arr[j-k];
            j-=k;
        }
        arr[j]=val;
    }
}

原地排序
我对冒泡排序的理解：一种简单的排序方式，对于一个数组的排序，从第一个元素开始，向后依次相邻元相互比较，将较大的那个元素放在后面，这样，一次循环结束，数组尾就是数组的最大元素。使用同样的方法，将第二大的元素移动至数组倒数第二的位置。如此循环，最后数组有序。

C代码如下：
#include<stdio.h>
  
void sort(int arr[],int n)
{
    int i,j,temp;
    for(i=0;i<n-1;i++)
    {
        for(j=0;j<n-i-1;j++)
        {
            if(arr[j]>arr[j+1])
            {
                temp=arr[j];
                arr[j]=arr[j+1];
                arr[j+1]=temp;
            }
        }
    }
}

原地排序。
从数组第二个元素开始，与前面元素进行比较，如果前面元素比其大，则将此比其大的元素后移，直到遇到比起小的第一个元素，并将此元素“插入”到这个比其小的第一个元素后面。

原地排序。
遍历数组，找出最小元素，与数组第一个元素交换。再从数组第二个元素开始遍历数组，找出最小元素，并与数组第二个元素交换。如此往复，最后数组有序。

原地排序。
桶排序的大体思路就是先将数组分到有限个桶中，再对每个桶中的数据进行排序，可以说是鸽巢排序的一种归纳结果（对每个桶中数据的排序可以是桶排序的递归，或其他算法，在桶中数据较少的时候用插入排序最为理想）
C代码：
#include<stdio.h>
#include<stdlib.h>
#include<string.h>

#define NUM 10

struct list_s
{
    int data;
    struct list_s * next;
};

void sort_bucket(int arr[],int n)
{
    int i,k=0;
    int max=arr[0];
    int min=arr[0];
    struct list_s *p,*s;
    for(i=0;i<n;i++)
    {   
        if(max<arr[i])
            max=arr[i];
        if(min>arr[i])
            min=arr[i];
    }
    int len=(max-min)/NUM+1;
    struct list_s *ptr=(struct list_s*)malloc(sizeof(struct list_s)*len);
    for(i=0;i<len;i++)
    {
        ptr[i].data=0;
        ptr[i].next=NULL;
    }
    for(i=0;i<n;i++)
    {
        int pos=(arr[i]-min)/NUM;
        struct list_s* node;
        node=(struct list_s *)malloc(sizeof(struct list_s));
        if(!node)
            return ;
        node->data=arr[i];
        node->next=NULL;
        if(ptr[pos].next==NULL)
        {
            ptr[pos].next=node;
        }
        else
        {
            for(p=&ptr[pos];p->next!=NULL;p=p->next)
            {
                if(p->next->data>node->data)
                {
                    node->next=p->next;
                    p->next=node;
                    break;
                }
            }
            p->next=node;
        }
    }
    for(i=0;i<len;i++)
    {
        for(p=&ptr[i];p->next!=NULL;p=p->next)
        {
            arr[k++]=p->next->data;
        }
    }
    for(i=0;i<len;i++)
    {
        p=&ptr[i];
        p=p->next;
        while(p)
        {
            s=p->next;
            free(p);
            p=s;
        }
    }
    free(ptr);
}


int main()
{
    int arr[]={45,45,13,29,21,25,24,11,39,38,35,37,44,48,41,66,63,99,91,93,88,89,83};
    sort_bucket(arr,23);
    int i;
    for(i=0;i<23;i++)
    {
        printf("%d ",arr[i]);
    }
    putchar('\n');
    return 0;
}

稳定排序，非原地排序
对于N个待排数据，M个桶，平均每个桶[N/M]个数据的桶排序平均时间复杂度为：O(N)+O(M*(N/M)*log(N/M))=O(N+N*(logN-logM))=O(N+N*logN-N*logM)，当N=M时，即极限情况下每个桶只有一个数据时。桶排序的最好效率能够达到O(N)。
是一种非比较排序算法 。它的主要思路是，
1. 将所有待排序整数（注意，必须是非负整数）统一为位数相同的整数，位数较少的前面补零。一般用10进制，也可以用16进制甚至2进制。所以前提是能够找到最大值，得到最长的位数，设 k 进制下最长为位数为 d 。
2. 从最低位开始，依次进行一次稳定排序。这样从最低位一直到最高位排序完成以后，整个序列就变成了一个有序序列。
举个例子，有一个整数序列，0, 123, 45, 386, 106，下面是排序过程：

第一次排序，个位，000 123 045 386 106，无任何变化
第二次排序，十位，000 106 123 045 386
第三次排序，百位，000 045 106 123 386
最终结果，0, 45, 106, 123, 386, 排序完成。

void sort(int arr[],int n)
{
    int *ptr,*count_arr,j,i,max,k=0;
    ptr=(int *)malloc(sizeof(int)*n);
    if(!ptr)
        return ;
    count_arr=(int *)malloc(sizeof(int)*10);
    if(!count_arr)
        return ;
    max=arr[0];
    for(i=0;i<n;i++)
    {
        if(max<arr[i])
            max=arr[i];
    }


    while(max)
    {
        max=max/10;
        k++;
    }
    int pos=1;
    for(i=k;i>0;i--)
    {
        memset(count_arr,0,sizeof(int)*10);
        for(j=0;j<n;j++)
        {
            count_arr[arr[j]/pos%10]++;
        }
        for(j=1;j<10;j++)
        {
            count_arr[j]+=count_arr[j-1];
        }
        for(j=n-1;j>=0;j--)
        {
            ptr[--count_arr[arr[j]/pos%10]]=arr[j];
        }
        for(j=0;j<n;j++)
        {
            arr[j]=ptr[j];
        }
        pos*=10;
    }
}

稳定排序
我觉得它也可以叫二叉树排序，利用二叉树的性质，对数组进行排序。堆排序将数组看作为一个完全二叉树，元素排列和二叉树节点排列一致，从最后一个非叶子节点开始进行最大值search，与其子树比较交换，递归直至根节点成为最大值，并将此值与数组最后一个元素交换，则数组最大值到了数组最后。如此往复，最后数组有序。
C代码实现：
#include<stdio.h>

void swap(int *a,int *b)
{
    int temp=*a;
    *a=*b;
    *b=temp;
}

void create_heap(int arr[],int node,int n)
{
    int left_child=node*2+1,right_child=node*2+2;
    int max_node=node;
    if(left_child<n && arr[max_node]<arr[left_child])
        max_node=left_child;
    if(right_child<n && arr[max_node]<arr[right_child])
        max_node=right_child;
    if(max_node!=node)
    {
        swap(&arr[max_node],&arr[node]);
        create_heap(arr,max_node,n);
    }
}


void sort_heap(int arr[],int n)
{
    int i,j;
    for(i=(n-2)/2;i>=0;i--)
        create_heap(arr,i,n);
    swap(&arr[0],&arr[n-1]);
    for(j=n-1;j>=1;j--)
        sort_heap(arr,j);
}

        printf("%d  ",arr[i]);

归并排序很好描述：将数组从中间切开，对左边和右边分别进行排序，最后数组左右都有序，此时合并数组进行排序会很简单。
C代码实现：
#include<stdio.h>
#include<stdlib.h>

void array(int arr[],int left,int mid,int right)
{
    int i=left,j=mid+1,k=0;
    int *ptr=(int *)malloc(sizeof(int)*(right-left+1));
    if(!ptr)
        return ;
    while(i<=mid && j<=right)
    {
        if(arr[i]<arr[j])
            ptr[k++]=arr[i++];
        else
            ptr[k++]=arr[j++];
    }
    while(i<=mid)
        ptr[k++]=arr[i++];
    while(j<=right)
        ptr[k++]=arr[j++];
    k=0;
    for(i=left;i<=right;i++)
    {
        arr[i]=ptr[k++];
    }
}


void sort(int arr[],int left,int right)
{
    if(left<right)
    {
        int mid=(left+right)/2;
        sort(arr,left,mid);
        sort(arr,mid+1,right);
        array(arr,left,mid,right);
    }
}

简单说明一下快速排序的原理：从数组中选出一个中轴元素，对数组进行排序，使数组左边元素不大于中轴元素，右边元素不小于中轴元素，并对数组左右再次尽心此选举和排序，最后数组有序。
C代码实现：
#include<stdio.h>

int sel_mid(int arr[],int left,int right)
{
    int mid=arr[left];
    int i=left+1,j=right;
    while(1)
    {
        while(i<=j && mid>=arr[i])
            i++;
        while(i<=j && mid<=arr[j])
            j--;
        if(i>=j)
            break;
        int temp=arr[i];
        arr[i]=arr[j];
        arr[j]=temp;
    }
    arr[left]=arr[j];
    arr[j]=mid;
    return j;
}


void sort(int arr[],int left,int right)
{
    if(left<right)
    {
        int mid=sel_mid(arr,left,right);
        sort(arr,left,mid-1);
        sort(arr,mid+1,right);
    }
}

原地排序
在算法中会经常用到这种方法，对于数组中最大值和最小值间隔不是很大的数组，采用计数排序会很简单。
找出数组中最大元素，创建一个数组最大元素值大小的数组，将原数组中的元素值对应成创建数组的下标并使其加1，最后根据创建数组中元素的值，以下标对应为元素的方式再对应到原数组中，使数组有序。
C代码如下：
#include<stdio.h>
#include<string.h>
#include<stdlib.h>

void sort(int arr[],int n)
{
    int i,j,k=0;
    int max=arr[0];
    for(i=0;i<n;i++)
    {
        if(max<arr[i])
            max=arr[i];
    }
    int *ptr=(int *)malloc(sizeof(int)*(max+1));
    if(!ptr)
        return ;
    for(i=0;i<max+1;i++)
        ptr[i]=-65536;
    for(i=0;i<n;i++)
    {
        if(ptr[arr[i]]==-65536)
        {
            ptr[arr[i]]=0;
        }
        ptr[arr[i]]++;
    }
    for(i=0;i<max+1;i++)
    {
        if(ptr[i]>=0)
        {
            for(j=0;j<ptr[i];j++)
            {
                arr[k++]=i;
            }
        }
    }
    return ;
}

#include<stdio.h>

        ptr[arr[i]-min]++;
    }
    for(i=0;i<max-min+1;i++)
    {
        if(ptr[i]>=0)
        {
            for(j=0;j<ptr[i];j++)
            {
                arr[k++]=i+min;
            }
        }
    }
    return ;
}
